def initialize(setting, locked: false, &validation)
  setting = setting.to_a[0]
  @name = setting[0]
  @value = setting[1]
  @locked = locked
  @value.freeze if @locked
  @initial_value = value.freeze
  @validation = block_given? ? validation : proc { true }
end
